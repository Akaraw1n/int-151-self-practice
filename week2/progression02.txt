ข้อที่ // 4.Constructor Function + for loop 
function Book(title,page){
    this.title = title
    this.page = page

} 
const books = [new Book("JS",345),new Book("html",245),new Book("Database",156)]
let info = []
for(let i = 0; i < books.length;i++){
    info.push(`${books[i].title} : ${books[i].page}`)
}
ข้อนี้จะเป็นการสร้างตัวแปร 1 ตัวที่เก็บ object 3 ตัวและให้แสดงออกมาในรูปของ array โดยการใช้ for-loop และ .push
